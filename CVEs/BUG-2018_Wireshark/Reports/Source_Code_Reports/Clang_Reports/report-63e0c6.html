359/*
360 Perform a DES encryption with a 16-byte key and 8-byte data item.
361 It's in fact 3 susbsequent call to crypt_des_ecb with a 7-byte key.
362 Missing bytes for the key are replaced by 0;
363 Returns output in response, which is expected to be 24 bytes.
364*/
365static int
366crypt_des_ecb_long(guint8 *response,
367 const guint8 *key,
368 const guint8 *data)
369{
370 guint8 pw21[21]; /* 21 bytes place for the needed key */
371
372 memset(pw21, 0, sizeof(pw21));
373 memcpy(pw21, key, 16);
374
375 memset(response, 0, 24);
376 /* crypt_des_ecb(data, key)*/
377 crypt_des_ecb(response, data, pw21, 1);
378 crypt_des_ecb(response + 8, data, pw21 + 7, 1);
379 crypt_des_ecb(response + 16, data, pw21 + 14, 1);
380
381 return 1;
382}

384/*
385 Generate a challenge response, given an eight byte challenge and
386 either the NT or the Lan Manager password hash (16 bytes).
387 Returns output in response, which is expected to be 24 bytes.
388*/
389static int
390ntlmssp_generate_challenge_response(guint8 *response,
391 const guint8 *passhash,
392 const guint8 *challenge)
393{
394 guint8 pw21[21]; /* Password hash padded to 21 bytes */
395
396 memset(pw21, 0x0, sizeof(pw21));
397 memcpy(pw21, passhash, 16);
398
399 memset(response, 0, 24);
400
401 crypt_des_ecb(response, challenge, pw21, 1);
402 crypt_des_ecb(response + 8, challenge, pw21 + 7, 1);
403 crypt_des_ecb(response + 16, challenge, pw21 + 14, 1);
404
405 return 1;
406}

426/* This function generate the Key Exchange Key
427 * Depending on the flags this key will either be used to crypt the exported session key
428 * or will be used directly as exported session key.
429 * Exported session key is the key that will be used for sealing and signing communication*/
430
431static void
432get_keyexchange_key(unsigned char keyexchangekey[NTLMSSP_KEY_LEN16], const unsigned char sessionbasekey[NTLMSSP_KEY_LEN16], const unsigned char lm_challenge_response[24], int flags)
433{
434 guint8 basekey[NTLMSSP_KEY_LEN16];
435 guint8 zeros[24];
436
437 memset(keyexchangekey, 0, NTLMSSP_KEY_LEN16);
438 memset(basekey, 0, NTLMSSP_KEY_LEN16);
439 /* sessionbasekey is either derived from lm_password_hash or from nt_password_hash depending on the key type negotiated */
440 memcpy(basekey, sessionbasekey, 8);
441 memset(basekey, 0xBD, 8);
442 if (flags&NTLMSSP_NEGOTIATE_LM_KEY0x00000080) {
443 /*data, key*/
444 crypt_des_ecb(keyexchangekey, lm_challenge_response, basekey, 1);
445 crypt_des_ecb(keyexchangekey+8, lm_challenge_response, basekey+7, 1);
446 }
447 else {
448 if (flags&NTLMSSP_REQUEST_NON_NT_SESSION0x00400000) {
449 /*People from samba tends to use the same function in this case than in the previous one but with 0 data
450 * it's not clear that it produce the good result
451 * memcpy(keyexchangekey, lm_hash, 8);
452 * Let's trust samba implementation it mights seem weird but they are more often rights than the spec !
453 */
454 memset(zeros, 0, 24);
455 crypt_des_ecb(keyexchangekey, zeros, basekey, 3);
456 crypt_des_ecb(keyexchangekey+8, zeros, basekey+7, 1);
457 }
458 else {
459 /* it is stated page 65 of NTLM SSP spec that sessionbasekey should be encrypted with hmac_md5 using the concact of both challenge
460 * when it's NTLM v1 + extended security but it turns out to be wrong !
461 */
462 memcpy(keyexchangekey, sessionbasekey, NTLMSSP_KEY_LEN16);
463 }
464 }
465}

522 */
523static void
524create_ntlmssp_v2_key(const char *nt_password _U___attribute__((unused)), const guint8 *serverchallenge , const guint8 *clientchallenge ,
525 guint8 *sessionkey , const guint8 *encryptedsessionkey , int flags ,
526 const ntlmssp_blob *ntlm_response, const ntlmssp_blob *lm_response _U___attribute__((unused)), ntlmssp_header_t *ntlmssph)
527{
528/* static const would be nicer, but -Werror=vla does not like it */
529#define DOMAIN_NAME_BUF_SIZE512 512
530#define USER_BUF_SIZE256 256
531#define BUF_SIZE(512 + 256) (DOMAIN_NAME_BUF_SIZE512 + USER_BUF_SIZE256)
532 char domain_name_unicode[DOMAIN_NAME_BUF_SIZE512];
533 char user_uppercase[USER_BUF_SIZE256];
534 char buf[BUF_SIZE(512 + 256)];
535 /*guint8 md4[NTLMSSP_KEY_LEN];*/
536 unsigned char nt_password_hash[NTLMSSP_KEY_LEN16];
537 unsigned char nt_proof[NTLMSSP_KEY_LEN16];
538 unsigned char ntowf[NTLMSSP_KEY_LEN16];
539 guint8 sessionbasekey[NTLMSSP_KEY_LEN16];
540 guint8 keyexchangekey[NTLMSSP_KEY_LEN16];
541 guint8 lm_challenge_response[24];
542 guint32 i;
543 guint32 j;
544 rc4_state_struct rc4state;
545 size_t user_len;
546 size_t domain_len;
547 md4_pass *pass_list = NULL((void*)0);
548 guint32 nb_pass = 0;
549 gboolean found = FALSE(0);
550
551 /* We are going to try password encrypted in keytab as well, it's an idea of Stefan Metzmacher <metze@samba.org>
552 * The idea is to be able to test all the key of domain in once and to be able to decode the NTLM dialogs */
553
554 memset(sessionkey, 0, NTLMSSP_KEY_LEN16);
555#if defined(HAVE_HEIMDAL_KERBEROS) || defined(HAVE_MIT_KERBEROS)
556 nb_pass = get_md4pass_list(&pass_list, nt_password);
557#endif
558 i = 0;
559 memset(user_uppercase, 0, USER_BUF_SIZE256);
560 user_len = strlen(ntlmssph->acct_name);
561 if (user_len < USER_BUF_SIZE256 / 2) {
562 memset(buf, 0, BUF_SIZE(512 + 256));
563 str_to_unicode(ntlmssph->acct_name, buf);
564 for (j = 0; j < (2*user_len); j++) {
565 if (buf[j] != '\0') {
566 user_uppercase[j] = g_ascii_toupper(buf[j]);
567 }
568 }
569 }
570 else {
571 /* Unable to calculate the session not enough space in buffer, note this is unlikely to happen but ......*/
572 return;
573 }
574 domain_len = strlen(ntlmssph->domain_name);
575 if (domain_len < DOMAIN_NAME_BUF_SIZE512 / 2) {
576 str_to_unicode(ntlmssph->domain_name, domain_name_unicode);
577 }
578 else {
579 /* Unable to calculate the session not enough space in buffer, note this is unlikely to happen but ......*/
580 return;
581 }
582 while (i < nb_pass) {
583 #ifdef DEBUG_NTLMSSP
584 fprintf(stderr, "Turn %d, ", i);
585 #endif
586 memcpy(nt_password_hash, pass_list[i].md4, NTLMSSP_KEY_LEN16);
587 printnbyte(nt_password_hash, NTLMSSP_KEY_LEN16, "Current NT password hash: ", "\n");
588 i++;
589 /* ntowf computation */
590 memset(buf, 0, BUF_SIZE(512 + 256));
591 memcpy(buf, user_uppercase, user_len*2);
592 memcpy(buf+user_len*2, domain_name_unicode, domain_len*2);
593 md5_hmac(buf, domain_len*2+user_len*2, nt_password_hash, NTLMSSP_KEY_LEN16, ntowf);
594 printnbyte(ntowf, NTLMSSP_KEY_LEN16, "NTOWF: ", "\n");
595
596 /* LM response */
597 memset(buf, 0, BUF_SIZE(512 + 256));
598 memcpy(buf, serverchallenge, 8);
599 memcpy(buf+8, clientchallenge, 8);
600 md5_hmac(buf, NTLMSSP_KEY_LEN16, ntowf, NTLMSSP_KEY_LEN16, lm_challenge_response);
601 memcpy(lm_challenge_response+NTLMSSP_KEY_LEN16, clientchallenge, 8);
602 printnbyte(lm_challenge_response, 24, "LM Response: ", "\n");
603
604 /* NT proof = First NTLMSSP_KEY_LEN bytes of NT response */
605 memset(buf, 0, BUF_SIZE(512 + 256));
606 memcpy(buf, serverchallenge, 8);
607 memcpy(buf+8, ntlm_response->contents+NTLMSSP_KEY_LEN16, MIN(BUF_SIZE - 8, ntlm_response->length-NTLMSSP_KEY_LEN)((((512 + 256) - 8) < (ntlm_response->length-16)) ? ((512
+ 256) - 8) : (ntlm_response->length-16)));
608 md5_hmac(buf, ntlm_response->length-8, ntowf, NTLMSSP_KEY_LEN16, nt_proof);
609 printnbyte(nt_proof, NTLMSSP_KEY_LEN16, "NT proof: ", "\n");
610 if (!memcmp(nt_proof, ntlm_response->contents, NTLMSSP_KEY_LEN16)) {
611 found = TRUE(!(0));
612 break;
613 }
614 }
615 if (!found) {
616 return;
617 }
618
619 md5_hmac(nt_proof, NTLMSSP_KEY_LEN16, ntowf, NTLMSSP_KEY_LEN16, sessionbasekey);
620 get_keyexchange_key(keyexchangekey, sessionbasekey, lm_challenge_response, flags);
621 /* now decrypt session key if needed and setup sessionkey for decrypting further communications */
622 if (flags & NTLMSSP_NEGOTIATE_KEY_EXCH0x40000000)
623 {
624 memcpy(sessionkey, encryptedsessionkey, NTLMSSP_KEY_LEN16);
625 crypt_rc4_init(&rc4state, keyexchangekey, NTLMSSP_KEY_LEN16);
626 crypt_rc4(&rc4state, sessionkey, NTLMSSP_KEY_LEN16);
627 }
628 else
629 {
630 memcpy(sessionkey, keyexchangekey, NTLMSSP_KEY_LEN16);
631 }
632
633 memcpy(ntlmssph->session_key, sessionkey, NTLMSSP_KEY_LEN16);
634}

636 /* Create an NTLMSSP version 1 key
637 * That is more complicated logic and methods and user challenge as well.
638 * password points to the ANSI password to encrypt, challenge points to
639 * the 8 octet challenge string
640 */
641static void
642create_ntlmssp_v1_key(const char *nt_password, const guint8 *serverchallenge, const guint8 *clientchallenge,
643 guint8 *sessionkey, const guint8 *encryptedsessionkey, int flags,
644 const guint8 *ref_nt_challenge_response, const guint8 *ref_lm_challenge_response,
645 ntlmssp_header_t *ntlmssph)
646{
647 unsigned char lm_password_upper[NTLMSSP_KEY_LEN16];
648 unsigned char lm_password_hash[NTLMSSP_KEY_LEN16];
649 unsigned char nt_password_hash[NTLMSSP_KEY_LEN16];
650 unsigned char challenges_hash[NTLMSSP_KEY_LEN16];
651 unsigned char challenges_hash_first8[8];
652 unsigned char challenges[NTLMSSP_KEY_LEN16];
653 guint8 md4[NTLMSSP_KEY_LEN16];
654 guint8 nb_pass = 0;
655 guint8 sessionbasekey[NTLMSSP_KEY_LEN16];
656 guint8 keyexchangekey[NTLMSSP_KEY_LEN16];
657 guint8 lm_challenge_response[24];
658 guint8 nt_challenge_response[24];
659 rc4_state_struct rc4state;
660 md5_state_t md5state;
661 char nt_password_unicode[256];
662 size_t password_len;
663 unsigned int i;
664 gboolean found = FALSE(0);
665 md4_pass *pass_list = NULL((void*)0);
666
667 static const unsigned char lmhash_key[] =
668 {0x4b, 0x47, 0x53, 0x21, 0x40, 0x23, 0x24, 0x25};
669
670 memset(sessionkey, 0, NTLMSSP_KEY_LEN16);
671 memset(lm_password_upper, 0, sizeof(lm_password_upper));
672 /* lm auth/lm session == (!NTLM_NEGOTIATE_NT_ONLY && NTLMSSP_NEGOTIATE_LM_KEY) || ! (EXTENDED_SECURITY) || ! NTLMSSP_NEGOTIATE_NTLM*/
673 /* Create a Lan Manager hash of the input password */
674 if (nt_password[0] != '\0') {
675 password_len = strlen(nt_password);
676 /*Do not forget to free nt_password_nt*/
677 str_to_unicode(nt_password, nt_password_unicode);
678 crypt_md4(nt_password_hash, nt_password_unicode, password_len*2);
679 /* Truncate password if too long */
680 if (password_len > NTLMSSP_KEY_LEN16)
681 password_len = NTLMSSP_KEY_LEN16;
682 for (i = 0; i < password_len; i++) {
683 lm_password_upper[i] = g_ascii_toupper(nt_password[i]);
684 }
685 }
686 else
687 {
688 /* Unable to calculate the session key without a password ... and we will not use one for a keytab*/
689 if (!(flags & NTLMSSP_NEGOTIATE_EXTENDED_SECURITY0x00080000)) {
690 return;
691 }
692 }
693 if ((flags & NTLMSSP_NEGOTIATE_LM_KEY0x00000080 && !(flags & NTLMSSP_NEGOTIATE_NT_ONLY0x00000400)) || !(flags & NTLMSSP_NEGOTIATE_EXTENDED_SECURITY0x00080000) || !(flags & NTLMSSP_NEGOTIATE_NTLM0x00000200)) {
694 crypt_des_ecb(lm_password_hash, lmhash_key, lm_password_upper, 1);
695 crypt_des_ecb(lm_password_hash+8, lmhash_key, lm_password_upper+7, 1);
696 ntlmssp_generate_challenge_response(lm_challenge_response,
697 lm_password_hash, serverchallenge);
698 memcpy(sessionbasekey, lm_password_hash, NTLMSSP_KEY_LEN16);
699 }
700 else {
701
702 memset(lm_challenge_response, 0, 24);
703 if (flags & NTLMSSP_NEGOTIATE_EXTENDED_SECURITY0x00080000) {
704#if defined(HAVE_HEIMDAL_KERBEROS) || defined(HAVE_MIT_KERBEROS)
705 nb_pass = get_md4pass_list(&pass_list, nt_password);
706#endif
707 i = 0;
708 while (i < nb_pass) {
709 /*fprintf(stderr, "Turn %d, ", i);*/
710 memcpy(nt_password_hash, pass_list[i].md4, NTLMSSP_KEY_LEN16);
711 /*printnbyte(nt_password_hash, NTLMSSP_KEY_LEN, "Current NT password hash: ", "\n");*/
712 i++;
713 if(clientchallenge){
714 memcpy(lm_challenge_response, clientchallenge, 8);
715 }
716 md5_init(&md5state);
717 md5_append(&md5state, serverchallenge, 8);
718 md5_append(&md5state, clientchallenge, 8);
719 md5_finish(&md5state, challenges_hash);
720 memcpy(challenges_hash_first8, challenges_hash, 8);
721 crypt_des_ecb_long(nt_challenge_response, nt_password_hash, challenges_hash_first8);
722 if (ref_nt_challenge_response && !memcmp(ref_nt_challenge_response, nt_challenge_response, 24)) {
723 found = TRUE(!(0));
724 break;
725 }
726 }
727 }
728 else {
729 crypt_des_ecb_long(nt_challenge_response, nt_password_hash, serverchallenge);
730 if (flags & NTLMSSP_NEGOTIATE_NT_ONLY0x00000400) {
731 memcpy(lm_challenge_response, nt_challenge_response, 24);
732 }
733 else {
734 crypt_des_ecb_long(lm_challenge_response, lm_password_hash, serverchallenge);
735 }
736 if (ref_nt_challenge_response &&
737 !memcmp(ref_nt_challenge_response, nt_challenge_response, 24) &&
738 ref_lm_challenge_response &&
739 !memcmp(ref_lm_challenge_response, lm_challenge_response, 24))
740 {
741 found = TRUE(!(0));
742 }
743 }
744 /* So it's clearly not like this that's put into NTLMSSP doc but after some digging into samba code I'm quite confident
745 * that sessionbasekey should be based md4(nt_password_hash) only in the case of some NT auth
746 * Otherwise it should be lm_password_hash ...*/
747 crypt_md4(md4, nt_password_hash, NTLMSSP_KEY_LEN16);
748 if (flags & NTLMSSP_NEGOTIATE_EXTENDED_SECURITY0x00080000) {
749 memcpy(challenges, serverchallenge, 8);
750 if(clientchallenge){
751 memcpy(challenges+8, clientchallenge, 8);
752 }
753 /*md5_hmac(text, text_len, key, key_len, digest);*/
754 md5_hmac(challenges, NTLMSSP_KEY_LEN16, md4, NTLMSSP_KEY_LEN16, sessionbasekey);
755 }
756 else {
757 memcpy(sessionbasekey, md4, NTLMSSP_KEY_LEN16);
758 }
759 }
760
761 if (!found) {
762 return;
763 }
764
765 get_keyexchange_key(keyexchangekey, sessionbasekey, lm_challenge_response, flags);
766 memset(sessionkey, 0, NTLMSSP_KEY_LEN16);
767 /*printnbyte(nt_challenge_response, 24, "NT challenge response", "\n");
768 printnbyte(lm_challenge_response, 24, "LM challenge response", "\n");*/
769 /* now decrypt session key if needed and setup sessionkey for decrypting further communications */
770 if (flags & NTLMSSP_NEGOTIATE_KEY_EXCH0x40000000)
771 {
772 if(encryptedsessionkey){
773 memcpy(sessionkey, encryptedsessionkey, NTLMSSP_KEY_LEN16);
774 }
775 crypt_rc4_init(&rc4state, keyexchangekey, NTLMSSP_KEY_LEN16);
776 crypt_rc4(&rc4state, sessionkey, NTLMSSP_KEY_LEN16);
777 }
778 else
779 {
780 memcpy(sessionkey, keyexchangekey, NTLMSSP_KEY_LEN16);
781 }
782 memcpy(ntlmssph->session_key, sessionkey, NTLMSSP_KEY_LEN16);
783}

871/* Create an NTLMSSP version 1 key.
872 * password points to the ANSI password to encrypt, challenge points to
873 * the 8 octet challenge string, key128 will do a 128 bit key if set to 1,
874 * otherwise it will do a 40 bit key. The result is stored in
875 * sspkey (expected to be NTLMSSP_KEY_LEN octets)
876 */
877/* dissect a string - header area contains:
878 two byte len
879 two byte maxlen
880 four byte offset of string in data area
881 The function returns the offset at the end of the string header,
882 but the 'end' parameter returns the offset of the end of the string itself
883 The 'start' parameter returns the offset of the beginning of the string
884 If there's no string, just use the offset of the end of the tvb as start/end.
885*/
886static int
887dissect_ntlmssp_string (tvbuff_t *tvb, int offset,
888 proto_tree *ntlmssp_tree,
889 gboolean unicode_strings,
890 int string_hf, int *start, int *end,
891 const char **stringp)
892{
893 proto_tree *tree = NULL((void*)0);
894 proto_item *tf = NULL((void*)0);
895 gint16 string_length = tvb_get_letohs(tvb, offset);
896 gint16 string_maxlen = tvb_get_letohs(tvb, offset+2);
897 gint32 string_offset = tvb_get_letohl(tvb, offset+4);
898 const char *string_text = NULL((void*)0);
899 int result_length;
900 guint16 bc;
901
902 *start = (string_offset > offset+8 ? string_offset : (signed)tvb_reported_length(tvb));
903 if (0 == string_length) {
904 *end = *start;
905 if (ntlmssp_tree)
906 proto_tree_add_string(ntlmssp_tree, string_hf, tvb,
907 offset, 8, "NULL");
908 if (stringp != NULL((void*)0))
909 *stringp = "";
910 return offset+8;
911 }
912
913 bc = result_length = string_length;
914 string_text = get_unicode_or_ascii_string(tvb, &string_offset,
915 unicode_strings, &result_length,
916 FALSE(0), TRUE(!(0)), &bc);
917
918 if (stringp != NULL((void*)0)) {
919 if (!string_text) string_text = ""; /* Make sure we don't blow up later */
920
921 *stringp = string_text;
922 }
923
924 if (ntlmssp_tree) {
925 tf = proto_tree_add_string(ntlmssp_tree, string_hf, tvb,
926 string_offset, result_length, string_text);
927 tree = proto_item_add_subtree(tf, ett_ntlmssp_string);
928 }
929 proto_tree_add_uint(tree, hf_ntlmssp_string_len,
930 tvb, offset, 2, string_length);
931 offset += 2;
932 proto_tree_add_uint(tree, hf_ntlmssp_string_maxlen,
933 tvb, offset, 2, string_maxlen);
934 offset += 2;
